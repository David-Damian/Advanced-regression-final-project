model {
# Likelihood
for (t in 1:(n+m)) {
    y[t] ~ dnorm(mu[t], eta1)
}
	
# State equations
for (t in 2:(n+m)) {
    mu[t] ~ dnorm(a[t], eta1)
    a[t] <- mu[t-1] + nu[t-1]
    nu[t] ~ dnorm(nu[t-1], eta2)
}

# Priors
mu[1] ~ dnorm(0,0.001)
nu[1] ~ dnorm(0,0.001)
eta1 ~ dgamma(0.001, 0.001)
eta2 ~ dgamma(0.001, 0.001)

#Prediction 1
for (i in 1:(n+m)) { yf1[i] ~ dnorm(mu[i],eta1) }
}
